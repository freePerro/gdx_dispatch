Quick Start

Install: pip install requests

import requests

API_KEY = "gdx_live_your_key_here"
BASE_URL = "https://{{ request.url.netloc }}/api/v1"

headers = {"X-API-Key": API_KEY}

# List jobs
resp = requests.get(f"{BASE_URL}/jobs", headers=headers, params={"page": 1, "per_page": 20})
print(resp.json())
# {"data": [...], "meta": {"page": 1, "per_page": 20, "total": 42}}

# Create a job
job = requests.post(f"{BASE_URL}/jobs", headers=headers, json={
    "title": "Spring tune-up",
    "status": "scheduled",
}).json()
print(job["data"]["id"])

# List customers
customers = requests.get(f"{BASE_URL}/customers", headers=headers).json()

# Create a customer
new_customer = requests.post(f"{BASE_URL}/customers", headers=headers, json={
    "name": "Jane Smith",
    "email": "jane@example.com",
    "phone": "555-0100",
}).json()

# List invoices (filter by status)
invoices = requests.get(
    f"{BASE_URL}/invoices",
    headers=headers,
    params={"status": "sent"},
).json()

# List jobs
curl -s "https://{{ request.url.netloc }}/api/v1/jobs?page=1&per_page=20" \
  -H "X-API-Key: gdx_live_your_key_here"

# Create a job
curl -s -X POST "https://{{ request.url.netloc }}/api/v1/jobs" \
  -H "X-API-Key: gdx_live_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{"title":"Spring tune-up","status":"scheduled"}'

# Update a job status
curl -s -X PATCH "https://{{ request.url.netloc }}/api/v1/jobs/JOB_ID" \
  -H "X-API-Key: gdx_live_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{"status":"in_progress"}'

# List customers (with search)
curl -s "https://{{ request.url.netloc }}/api/v1/customers?search=smith" \
  -H "X-API-Key: gdx_live_your_key_here"

# Register a webhook
curl -s -X POST "https://{{ request.url.netloc }}/api/v1/webhooks" \
  -H "X-API-Key: gdx_live_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{"url":"https://your-server.com/hooks/gdx","events":["job.created","job.updated"]}'
